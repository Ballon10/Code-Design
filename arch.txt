  var timeoutInMinutes = 5;
var timeoutId;

function startTimer() {
    clearTimeout(timeoutId);
    timeoutId = setTimeout(logoutUser, timeoutInMinutes * 60 * 1000); // Convert to milliseconds
}

function resetTimer() {
    clearTimeout(timeoutId);
    startTimer();
}

// Start the timer when the user interacts with the page
document.addEventListener('mousemove', resetTimer);
document.addEventListener('keypress', resetTimer);

function logoutUser() {
    // Redirect to logout script or clear session
    window.location.href = '../pages/logout.php';
}

// Start the timer when the page loads
startTimer();

</script>

<script>
document.getElementById('updateRebus').addEventListener('click', function() {
   // Empêche l'envoi du formulaire par défaut
   event.preventDefault();

    // Récupérer les valeurs du formulaire
    let pdf_file = document.getElementById('motif').value;
    let recuPar = document.getElementById('recuPar').value;
	const fichier = document.getElementById('pdf_file').files[0];
    // Récupère toutes les cases à cocher sélectionnées
    let selectedItems = document.querySelectorAll('.selectItem:checked');
    let ids = [];

    selectedItems.forEach(item => {
        ids.push(item.getAttribute('data-id'));
    });

    if (ids.length > 0) {
        // Créer un objet FormData pour envoyer les données
        let formData = new FormData();
        formData.append('motif', motif);
        formData.append('recuPar', recuPar);
		formData.append('pdf_file', fichier)
        formData.append('ids', JSON.stringify(ids)); // Envoyer les IDs sous forme JSON

        // Envoie les données au serveur via AJAX
        let xhr = new XMLHttpRequest();
        xhr.open('POST', 'approve_decharge.php', true);

        xhr.onload = function() {
            if (xhr.status === 200) {
                alert('Les éléments sélectionnés ont été approuvés avec succes.');
                location.reload(); // Recharger la page pour voir les modifications
            } else {
                alert('Erreur lors de l\'approbation des éléments sélectionnés');
            }
        };
        xhr.onerror = function() {
            alert('Une erreur est survenue lors de la connexion au serveur.');
        };

        xhr.send(formData);
    } else {
        alert('Veuillez sélectionner au moins un élément');
    }
});
</script>
